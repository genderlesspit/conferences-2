	on_startup = {
		effect = {
			every_country = {
				set_variable = {
					var = total_state_conservative
					value = 0
				}
				set_variable = {
					var = total_state_liberal
					value = 0
				}
				set_variable = {
					var = total_state_neutral
					value = 0
				}
				set_variable = {
					var = total_state_communist
					value = 0
				}
				set_variable = {
					var = total_state_fascist
					value = 0
				}			
			}
			every_country = {
				limit = {
					has_government = democratic
				}
				set_temp_variable = {
						var = governor_chance_democratic
						value = party_popularity_100@democratic	
				}
				set_temp_variable = {
					var = governor_chance_neutrality
					value = party_popularity_100@neutrality
				}
				set_temp_variable = {
					var = governor_chance_communism
					value = party_popularity_100@communism
				}
				set_temp_variable = {
					var = governor_chance_fascism
					value = party_popularity_100@fascism
				}
				every_owned_state = {
					limit = {
						is_core_of = PREV
					}
					random_list = {
						var:governor_chance_democratic = {
							random_list = {
								10 = {
									set_state_flag = db_state_conservative
									PREV = {
										add_to_variable = {
											var = total_state_conservative
											value = 1
											tooltip = add_total_state_conservative
										}
									}
									add_dynamic_modifier = {
										modifier = db_state_modifier_conservative
										scope = THIS
										
									}
									modifier = { factor = 2 PREV = { has_country_leader_ideology = conservatism } }
								}
								10 = {
									set_state_flag = db_state_liberal
									PREV = { 
										add_to_variable = {
											var = total_state_liberal
											value = 1
											tooltip = add_total_state_liberal
										}
									}
									add_dynamic_modifier = {
										modifier = db_state_modifier_liberal
										scope = THIS
										
									}
									modifier = { factor = 2 PREV = { has_country_leader_ideology = liberalism } }
								}
							}
						}
						var:governor_chance_neutrality = {
							set_state_flag = db_state_neutral
							PREV = { 
								add_to_variable = {
									var = total_state_neutral
									value = 1
									tooltip = add_total_state_neutral
								}
							}
							add_dynamic_modifier = {
								modifier = db_state_modifier_neutral
								scope = THIS
								
							}
						}	
						var:governor_chance_communism = {
							set_state_flag = db_state_communist
							PREV = {
								add_to_variable = {
									var = total_state_communist
									value = 1
									tooltip = add_total_state_communist
								}
							}
							add_dynamic_modifier = {
								modifier = db_state_modifier_communist
								scope = THIS
								
							}
						}
						var:governor_chance_fascism = {
							set_state_flag = db_state_fascist
							PREV = {
								add_to_variable = {
									var = total_state_fascist
									value = 1
									tooltip = add_total_state_fascist
								}
							}
							add_dynamic_modifier = {
								modifier = db_state_modifier_fascist
								scope = THIS
								
							}
						}
					}
				}
			}
		}
	}

	on_weekly = {
		effect = {
			every_country = {
				set_variable = {
					var = db_pp_conservative
					value = 0
				}
				set_variable = {
					var = db_pp_liberal
					value = 0
				}
				set_variable = {
					var = db_pp_neutral
					value = 0
				}
				set_variable = {
					var = db_pp_communist
					value = 0
				}
				set_variable = {
					var = db_pp_fascist
					value = 0
				}
				for_each_loop = {
					array = THIS.total_states_ideology
					value = x
					add_to_variable = {
						var = db_total_ideological_states
						value = x
					}
				}
				for_each_loop = {
					array = THIS.db_each_ideology_pp
					value = x
					add_to_variable = {
						var = db_pp_conservative
						value = x
					}
				}
				for_each_loop = {
					array = THIS.db_each_ideology_pp
					value = x
					divide_variable = {
						var = x
						value = db_total_ideological_states
					}
				}
				add_to_variable = {
					var = db_dem_faction_split
					value = party_popularity@democratic
				}
				divide_variable = {
					var = db_dem_faction_split
					value = 2
				}
				add_to_variable = {
					var = db_pp_conservative
					value = db_dem_faction_split
				}
				add_to_variable = {
					var = db_pp_liberal
					value = db_dem_faction_split
				}
				add_to_variable = {
					var = db_pp_neutral
					value = party_popularity@neutrality
				}
				add_to_variable = {
					var = db_pp_communist
					value = party_popularity@communism
				}
				add_to_variable = {
					var = db_pp_fascist
					value = party_popularity@fascism
				}
				for_each_loop = {
					array = THIS.db_each_ideology_pp
					value = x
					divide_variable = {
						var = x
						value = 2
					}
				}
				for_each_loop = {
					array = THIS.db_each_ideology_pp
					value = x
					multiply_variable = {
						var = x
						value = 100
					}
				}
			}
		}
	}
