country_event = {
	id = db_hidden.5
	hidden = yes
	is_triggered_only = yes
	immediate = {
		hidden_effect = {
			every_owned_state = {
			
			# CALCULATE STATE TOTAL DEBT/SAVINGS
				add_to_variable = { THIS.db_state_consumer_debt_savings = THIS.db_state_daily_net_income }
				
			# OVERFLOW CHECK
				if = { 
					limit = { 
						OR = {
							ROOT = { has_country_flag = db_economic_scale_million }
							check_variable = { THIS.db_state_consumer_debt_savings > 1000000 }
						}
					}
					if = {
						limit = { ROOT = { NOT = { has_country_flag = db_economic_scale_million } } }
						ROOT = { set_country_flag = db_economic_scale_million }
					}
					if = { 
						limit = { 
							OR = {
								ROOT = { 
									has_country_flag = db_economic_scale_million
									has_country_flag = db_economic_scale_billion 
								}
								check_variable = { THIS.db_state_consumer_debt_savings > 1000 }
							}
						}
						if = {
							limit = { ROOT = { NOT = { has_country_flag = db_economic_scale_billion } } }
							ROOT = { set_country_flag = db_economic_scale_billion }
						}
					}
				}
			
			# SET STATE PER CAPITA SAVINGS/DEBT
				set_variable = { THIS.db_state_consumer_debt_savings_per_capita = THIS.db_state_consumer_debt_savings }
				divide_variable = { THIS.db_state_consumer_debt_savings_per_capita = THIS.db_state_lfpr }
		
			# OVERFLOW CHECK
				if = { 
					limit = { ROOT = { has_country_flag = db_economic_scale_million } }
					multiply_variable = { THIS.db_state_consumer_debt_savings_per_capita = 1000000 }
					if = { 
						limit = { ROOT = { has_country_flag = db_economic_scale_billion } }
						multiply_variable = { THIS.db_state_consumer_debt_savings_per_capita = 1000 }
					}
				}
			}
		}
	}
}
